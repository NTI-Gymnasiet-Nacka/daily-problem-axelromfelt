# setup
1. create a .venv [ py -3 -m venv .venv ]
2. install requirements.txt [ pip install -r ./requirements.txt ]
3. initialize the data base for the server [ flask --app flaskr init-db ]

# running the program
[ flask --app flaskr run --host=0.0.0.0 ]
